234 *
235 * \fn vWRITE_COMMAND_TO_LCD(void)
236 *
237 * \brief Function to write commands to the LCD
238 *
239 * \details Function is write commands to the LCD using the read/write
240 * commands and the register select
241 *
242 * \params[in] RS, data
243 *
244 * \returns nothing
245 *
246 * Modification History:
247 *
248 * 11/17/2013 - Original Function
249 *
250 ******************************************************************************
251 */
252 void vWRITE_COMMAND_TO_LCD(char RS, char data)
253 {
254  if(RS == DATA_WR) LCP = 1 << LCD_RS; /*Set RS high to write data*/
255  else LCP = 0x00; /*Set RS low to write instructions*/
256 
258  LCP = LCP | 1<<LCD_E;
259 
261  _delay_us(50);
262 
263  /* Write Data*/
264  LDP = data;
265  if (RS = DATA_WR)
266  {
268  CURSOR_X_POSITION++;
269  }
270 
271  /*Set Read/Write low*/
272  LCP = LCP & 1<<LCD_RS;
273 
275  _delay_us(50);
276 }

504  * \fn vLCD_GO_TO_Position(uint8_t, uint8_t)
505  *
506  * \brief Function to directly control LCD cursor Position
507  *
508  * \details Function is called to move the LCD cursor to a specific location
509  * on the LCD.
510  *
511  * Allows for direct control of where data can be written, rather
512  * than only from the beginning of a line
513  *
514  * Saves current cursor position for text wrapping
515  *
516  * \params[in] Character, Row
517  *
518  * \returns nothing
519  *
520  * Modification History:
521  *
522  * 11/15/2013 - Original Function
523  *
524  ******************************************************************************
525  */
526 void vLCD_GO_TO_POSITION(uint8_t x, uint8_t y)
527 {
528  //save the current position
529  register uint8_t DDRAMAddr;
530 
531  // For each line o the LCD
532  switch(y)
533  {
534  case 0:
535  //for the top line, set the DDRAM address
536  //move a certain number of characters to the right
537  DDRAMAddr = LCD_LINE0_DDRAMADDR + x;
538  break;
539 
540  case 1:
541  //for the bottom line, set the DDRAM address
542  //move a certain number of characters to the right
543  DDRAMAddr = LCD_LINE1_DDRAMADDR + x;
544  break;
545 
546  default:
547  //default to the top left of the LCD if nothing is specified
548  DDRAMAddr = LCD_LINE0_DDRAMADDR+x;
549  }
550 
551  //save current cursor position X
552  CURSOR_X_POSITION = x;
553  //save current cursor position Y
554  CURSOR_Y_POSITION = y;
555 
556  // send a command to set the data address
557  vWRITE_COMMAND_TO_LCD(INSTR_WR, 1 <<LCD_DDRAM | DDRAMAddr);
558 
559  // Delay greater than 39us to allow LCD to complete operation
560  _delay_us(50);
561 }
